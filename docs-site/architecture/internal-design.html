Internal Design

Signal's internal graph structures and implementation details.

Signal Internal State

Source signals maintain minimal internal state:

Properties

  • _value - The current value (stored immutably via @cyftech/immutjs)
  • _effects - A Set of registered SignalsEffect functions

Implementation

// Simplified internal structure
interface SourceSignal<T> {
  type: "source-signal";
  _value: T;  // Stored immutably
  _effects: Set<SignalsEffect>;
  get value(): T;
  set value(newValue: T);
}

Derived Signal Internal State

Derived signals are implemented as a source signal + effect pattern:

Properties

  • oldValue - Previous computed value
  • derivedSource - Internal source signal holding the computed value
  • derivedSourceUpdator - Internal effect that recomputes the value

Implementation

// Simplified internal structure
interface DerivedSignal<T> {
  type: "derived-signal";
  oldValue: T | undefined;
  derivedSource: SourceSignal<T>;
  derivedSourceUpdator: SignalsEffect;
  get value(): T;
  get prevValue(): T | undefined;
  dispose(): void;
}

Effect Internal State

Effects maintain minimal state for disposal tracking:

Properties

  • canDisposeNow - Boolean flag for disposal status
  • dispose() - Method to mark for disposal

Implementation

// Simplified internal structure
interface SignalsEffect {
  canDisposeNow: boolean;
  dispose(): void;
}

Global Effect Tracking

A module-level variable tracks the currently executing effect:

// Module-level variable
let _currentSignalEffect: SignalsEffect | null = null;

// Set before executing effect
function effect(fn: () => void): SignalsEffect {
  const eff = fn as SignalsEffect;
  eff.canDisposeNow = false;
  
  _currentSignalEffect = eff;
  fn();
  _currentSignalEffect = null;
  
  return eff;
}

Signal Value Getter

The signal value getter checks for the current effect and registers dependencies:

// Simplified value getter
get value(): T {
  if (_currentSignalEffect) {
    this._effects.add(_currentSignalEffect);
  }
  return newVal(this._value);
}

Signal Value Setter

The signal value setter updates the value and triggers effects:

// Simplified value setter
set value(newValue: T) {
  if (newValue === this._value) return;
  
  this._value = immut(newValue);
  
  for (const eff of this._effects) {
    if (eff.canDisposeNow) {
      this._effects.delete(eff);
    } else {
      _currentSignalEffect = eff;
      eff();
      _currentSignalEffect = null;
    }
  }
}

Derived Signal Creation

Derived signals are created by wrapping a source signal + effect:

// Simplified derive implementation
function derive<T>(valueGetterFn: (oldValue: T | undefined) => T): DerivedSignal<T> {
  let oldValue: T | undefined = undefined;
  const derivedSource = signal(valueGetterFn(oldValue));
  
  const derivedSourceUpdator = effect(() => {
    const newValue = valueGetterFn(oldValue);
    oldValue = newValue;
    derivedSource.value = newValue;
  });
  
  return {
    type: "derived-signal",
    oldValue,
    derivedSource,
    derivedSourceUpdator,
    get value() { return derivedSource.value; },
    get prevValue() { return oldValue; },
    dispose() { derivedSourceUpdator.dispose(); }
  };
}

Array Signal Implementation

Array signals are extended with mutation methods that create new arrays internally:

// Simplified array mutation
function push<T>(this: SourceSignal<T[]>, ...items: T[]): void {
  const newArray = [...this.value, ...items];
  this.value = newArray; // Triggers effects
}

// Custom remove method
function remove<T>(this: SourceSignal<T[]>, predicate: (item: T) => boolean): void {
  const newArray = this.value.filter(item => !predicate(item));
  this.value = newArray; // Triggers effects
}

Object Signal Implementation

Object signals have a set() method for partial updates:

// Simplified set implementation
function set<T extends object>(this: SourceSignal<T>, partial: Partial<T>): void {
  const newValue = { ...this.value, ...partial };
  this.value = newValue; // Triggers effects
}
